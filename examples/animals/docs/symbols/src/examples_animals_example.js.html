 17 /*
 18  * Does not require an opts parameter as we are providing all the options in 
 19  * the closure.json file in this directory;
 20  */
 21 require( 'goog' ).goog.init();
 22 
 23 /*
 24  * Now that the nodegoog is initialised you can use any base.js functionality 
 25  * such as goog.require / goog.provide
 26  *
 27  * Note: At least one namespace must be provided for compilation purposes
 28  */
 29  
 30 goog.provide('node.goog.examples.animals.Example'); 
 31 
 32 goog.require('goog.array');
 33 goog.require('node.goog.examples.animals.IAnimal');
 34 goog.require('node.goog.examples.animals.Dog');
 35 goog.require('node.goog.examples.animals.Cat');
 36 goog.require('node.goog.examples.animals.Tiger');
 37 goog.require('node.goog.examples.animals.Monkey');
 38 
 39 /** 
 40  * @constructor
 41  */
 42 node.goog.examples.animals.Example = function() {
 43   /**
 44    * @private
 45    * @type {Array.<node.goog.examples.animals.IAnimal>}
 46    */
 47   this.animals_ = this.initRandomAnimals_();
 48   this.makeAnimalsTalk_();
 49 };
 50 
 51 /** 
 52  * @return {Array.<node.goog.examples.animals.IAnimal>} A colleciton of 
 53  *    random animals
 54  */
 55 node.goog.examples.animals.Example.prototype.initRandomAnimals_ = function() {
 56   /** @type {Array.<node.goog.examples.animals.IAnimal>} */
 57   var animals = [];
 58   var types = [
 59     node.goog.examples.animals.Dog, 
 60     node.goog.examples.animals.Cat, 
 61     node.goog.examples.animals.Tiger, 
 62     node.goog.examples.animals.Monkey
 63   ];
 64   var len = parseInt(10 + (Math.random() * 10), 10);
 65   console.log('Creating ' + len + ' random animals');
 66   for (var i = 0; i < len; i++) {
 67     var t = types[parseInt(Math.random() * types.length, 10)];    
 68     animals.push(new t());
 69   };
 70   return animals;
 71 };
 72 
 73 
 74 /** 
 75  * @private
 76  */
 77 node.goog.examples.animals.Example.prototype.makeAnimalsTalk_ = function() {
 78   /** @type {Array.<node.goog.examples.animals.IAnimal>} */
 79   goog.array.forEach(this.animals_, this.makeAnimalTalk_, this);
 80   
 81 };
 82 
 83 /** 
 84  * @private
 85  * @param {node.goog.examples.animals.IAnimal} animal
 86  */
 87 node.goog.examples.animals.Example.prototype.makeAnimalTalk_ = 
 88     function(animal) {
 89   animal.talk();  
 90 };
 91 
 92 
 93 /*
 94  * Start the demo
 95  */
 96 new node.goog.examples.animals.Example();
